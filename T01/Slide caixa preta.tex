\section{Seleção e Mutação}

\begin{frame}
\frametitle{Seleção e Mutação}
	\textbf{Critério:} Média das melhores soluções (Para cada uma das repetições realizadas determina-se a qualidade da melhor solução. Esta medida é obtida efetuando a média destes valores).
	\begin{itemize}
		\item \textbf{Torneio X Roleta}\\
		Resultado: A seleção por Torneio teve uma média ligeiramente maior do que a seleção por Roleta.
		\item \textbf{Bit a Bit X 1 Bit}\\
		Resultado: A mutação Bit a Bit teve uma média bem maior do que a mutação por 1 Bit, que na maioria das vezes ficava preso em um ótimo local.
	\end{itemize}
\end{frame}
